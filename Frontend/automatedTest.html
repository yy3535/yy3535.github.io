# js单元测试的一门语言Jest

npm install --save-dev jest

// sum.js
function sum(a, b) {
  return a + b;
}
module.exports = sum;

const sum = require('./sum');

test('adds 1 + 2 to equal 3', () => {
  expect(sum(1, 2)).toBe(3);
});

// package.json
{
  "scripts": {
    "test": "jest"
  }
}

PASS  ./sum.test.js
✓ adds 1 + 2 to equal 3 (5ms)
